package com.horacio.mutant.service;

import lombok.extern.log4j.Log4j2;
import org.springframework.stereotype.Service;
import org.springframework.util.StringUtils;

import java.util.HashMap;
import java.util.Map;

@Service
@Log4j2
public class MutantDetector4Letters implements MutantDetector{
    @Override
    public DetectionResult detectMutant(String[] dna) {
        return analyzeDna(dna);
    }

    private void validateDna(String[] dna){
        if (dna==null || dna.length == 0){
            throw new IllegalArgumentException();
        }
    }

    private void validateDnaRow(int rowSize, String dnaRow){
        if (StringUtils.isEmpty(dnaRow) || rowSize != dnaRow.length()){
            throw new IllegalArgumentException();
        }
    }

    private void validateChar(char dnaChar){
        if (dnaChar != 'A' && dnaChar != 'T' && dnaChar != 'C' && dnaChar != 'G'){
            throw new IllegalArgumentException();
        }
    }

    private DetectionResult analyzeDna(String[] dna){

        //TODO enable/disble logging
        log.debug("Analyzind Dna");

        Map<Integer, CharCount> verticalMatches = new HashMap<>(); // key = column, value = match count
        Map<Integer, CharCount> diagonal1MatchesCurrRow = new HashMap<>(); // key = column, value = match count
        Map<Integer, CharCount> diagonal1MatchesPrevRow = new HashMap<>();
        Map<Integer, CharCount> diagonal2MatchesCurrRow = new HashMap<>(); // key = column, value = match count
        Map<Integer, CharCount> diagonal2MatchesPrevRow = new HashMap<>();
        int horizontalMatches=0;

        char prevChar=' ';
        int rowSize = dna[0].length();
        int sequenceCount=0;

        for (int rowIndex=0; rowIndex<dna.length && sequenceCount<2; rowIndex++){
            // for each row verify size. should be >=4 and equal to the previous row
            String row = dna[rowIndex];
            validateDnaRow(rowSize, row);
            // TODO: check if this is optimo
            horizontalMatches=1;

            for (int colIndex=0; colIndex<row.length(); colIndex++){
                char currChar = row.charAt(colIndex);

                //TODO: tiene sentido hacer esto? xq si encuentra 4 letras iguales validas no va a validar el resto
                //validateChar(currChar);

                if (prevChar==currChar) {
                    horizontalMatches++;
                    if (horizontalMatches == 4) {
                        sequenceCount++;
                        if (sequenceCount==2){
                            break;
                        }
                        horizontalMatches=1;
                        prevChar=' ';
                    }
                }
                else{
                    horizontalMatches=1;
                }
                prevChar = currChar;


                CharCount verticalCount = verticalMatches.get(colIndex);
                if (verticalCount != null){
                    if (verticalCount.charFound == currChar) {
                        verticalCount.count++;
                        if (verticalCount.count == 4) {
                            sequenceCount++;
                            if (sequenceCount==2) {
                                break;
                            }
                            verticalCount.reset();
                        }
                    }
                }else{
                    verticalCount = new CharCount();
                    //verticalCount.count = 1;
                }
                verticalCount.charFound = currChar;
                verticalMatches.put(colIndex, verticalCount);

                CharCount diagonal1Count = diagonal1MatchesPrevRow.get(colIndex-1);
                if (diagonal1Count != null){
                    if (diagonal1Count.charFound == currChar) {
                        diagonal1Count.count++;
                        if (diagonal1Count.count == 4) {
                            sequenceCount++;
                            if (sequenceCount==2) {
                                break;
                            }
                            diagonal1Count.reset();
                        }
                    }
                }
                else{
                    diagonal1Count = new CharCount();
                    //diagonal1Count.count = 1;
                }
                diagonal1Count.charFound = currChar;
                diagonal1MatchesCurrRow.put(colIndex, diagonal1Count);


                CharCount diagonal2Count = diagonal2MatchesPrevRow.get(colIndex+1);
                if (diagonal2Count != null){
                    if (diagonal2Count.charFound == currChar) {
                        diagonal2Count.count++;
                        if (diagonal2Count.count == 4) {
                            sequenceCount++;
                            if (sequenceCount==2) {
                                break;
                            }
                            diagonal2Count.reset();
                        }
                    }
                }
                else{
                    diagonal2Count = new CharCount();
                    //diagonal2Count.count = 1;
                }
                diagonal2Count.charFound = currChar;
                diagonal2MatchesCurrRow.put(colIndex, diagonal2Count);
            }

            diagonal1MatchesPrevRow = diagonal1MatchesCurrRow;
            diagonal1MatchesCurrRow = new HashMap<>();

            diagonal2MatchesPrevRow = diagonal2MatchesCurrRow;
            diagonal2MatchesCurrRow = new HashMap<>();

        }

        //TODO improve this
        String completeDna = String.join("", dna);
        return new DetectionResult(sequenceCount==2, completeDna);
    }

    private class CharCount{
        private char charFound;
        private int count;

        public CharCount(){
            this.count = 1;
            charFound = ' ';
        }

        private void reset() {
            count = 1;
            charFound = ' ';
        }
    }
}
